def effect_est(data:pd.DataFrame, sign_col:list, value_cols:list)->float:
    """
    # Factor Effect Estimator for DOE Class.

    Parameters:
        data (pd.DataFrame): pandas dataframe with design matrix, and real number data.
        sign_col (list): A list of strings in design matrix in "`data`".
        value_cols (list): A list of strings real number data.

    Returns:
        Factor_Effect (pd.DataFrame): `(data[sign_col] * data[value_cols].sum(1)).mean()`

    """
    if isinstance(sign_col, list):
        return (data[sign_col].prod(1) * data[value_cols].sum(1)).mean()
    else: 
        return (data[sign_col] * data[value_cols].sum(1)).mean()
